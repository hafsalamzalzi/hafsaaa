package bibliothecaire;

public class biblio {
	
	
	    private Livre [] liste;
	    private int courant=0;
	    public biblio(int max){
	         liste =new Livre[max];
	       
	    }
	    public boolean addlivre(Livre L){
	        if (courant < liste.length){
	           this.liste[courant]=L;
	            courant++;
	            return true;
	        }
	        return false;
	    }
	    public boolean delLivre(Livre L){

	    for(int i=0;i<liste.length;i++){
	        
	        if(liste[i]==L){
	             for (int j = i; j <= liste.length - 1; j++)
	             liste[i]=liste[i+1];
	        }
	      }
	         courant--; 
	        return true;
	 }
	    	    public boolean updateLivre(Livre auncien,Livre L){

		    for(int i=0;i<liste.length;i++){
		        
		        if(liste[i]==auncien){
		             
		             liste[i]=L;
	                     
		        }
		      return true;
		        }
	        return false;
		 }
	            
	     public void affiche() {
	          for(Livre L:liste)
	          System.out.println("Livre [code:"+L.getCode()+", Titre:"+L.getTitre()+", Auteur: " + L.getAuteur());
	 }
	     }
